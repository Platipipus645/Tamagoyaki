11   class Vectorizer:
12       def __init__(self):
13           self.resnet_model = ResNet50(weights='imagenet', include_top=False, input_shape=(224, 224, 3))
14   
15       def transform(self, images: Sequence[str]) -> np.ndarray:
16           features = []
17           for img_path in images:
18               preprocess_img = self._preprocess_image(img_path)
19               if preprocess_img is not None:
20                   feature = self.resnet_model.predict(preprocess_img)
21                   flattened_feature = feature.flatten()
22                   features.append(flattened_feature)
23   
24           result = np.array(features)
25           print("Vectorization result shape:", result.shape)  # Debugging line
26           return result
27   
28       def _preprocess_image(self, img_path):
29           try:
30               img = image.load_img(img_path, target_size=(224, 224))
31               img_array = image.img_to_array(img)
32               img_array = np.expand_dims(img_array, axis=0)
33               img_array = preprocess_input(img_array)
34               return img_array
35           except Exception as e:
36               print(f"Error processing image at {img_path}: {e}")
37               return None

39   def cosine_similarity(query_image_vector: np.ndarray, image_vectors: np.ndarray) -> np.ndarray:
40       query_magnitude = np.linalg.norm(query_image_vector)
41       image_magnitudes = np.linalg.norm(image_vectors, axis=1)
42   
43       dot_products = np.dot(query_image_vector, image_vectors.T)
44       cos_similarity = dot_products / (query_magnitude * image_magnitudes)
45   
46       return cos_similarity

48   class ImageSearchEngine:
49       def __init__(self, vectorizer, image_vectors):
50           self.vectorizer = vectorizer
51           self.image_vectors = image_vectors
52   
53       def most_similar(self, query: np.ndarray, n: int = 5) -> List[Tuple[float, str]]:
54           query_vector = self.vectorizer.transform([query])
55           query_vector = np.reshape(query_vector, (1, -1))
56           print("Shape of query_vector before cosine_similarity:", query_vector.shape)  # Debugging line
57           similarities = cosine_similarity(query_vector, self.image_vectors)[0]
58   
59           top_indices = np.argsort(similarities)[::-1][:n]
60   
61           similar_images = [(similarities[index], f"Image_{index}") for index in top_indices]
62   
63           return similar_images

65       def visualize_similar_images(self, similar_images: List[Tuple[float, str]], image_db_path: str, query_image_path: str) -> None:
66           # Visualize the query image
67           plt.figure(figsize=(15, 3))
68           plt.subplot(1, len(similar_images) + 1, 1)
69           query_img = plt.imread(query_image_path)
70           plt.imshow(query_img)
71           plt.title("Query Image")
72           plt.axis('off')
73   
74           # Visualize similar images
75           for i, (similarity, image_name) in enumerate(similar_images):
76               plt.subplot(1, len(similar_images) + 1, i + 2)
77               img_path = os.path.join(image_db_path, f"{image_name}.jpg")
78               img = plt.imread(img_path)
79               plt.imshow(img)
80               plt.title(f"Similarity: {similarity:.2f}\nImage Name: {image_name}")
81               plt.axis('off')
82   
83           plt.show()

85       def create_collage(self, similar_images: List[Tuple[float, str]], image_db_path: str, output_path: str) -> None:
86           images = []
87           for _, image_name in similar_images:
88               img_path = os.path.join(image_db_path, f"{image_name}.jpg")
89               img = Image.open(img_path)
90               images.append(img)
91   
92           # Determine the dimensions of the collage (3xN grid)
93           width, height = images[0].size
94           collage_width = width * min(len(images), 3)
95           collage_height = height * ((len(images) - 1) // 3 + 1)
96   
97           # Create a new image for the collage
98           collage = Image.new('RGB', (collage_width, collage_height))
99   
100          # Paste images onto the collage
101          for i, img in enumerate(images):
102              collage.paste(img, (i % 3 * width, i // 3 * height))
103  
104          # Save the collage
105          collage.save(output_path)
